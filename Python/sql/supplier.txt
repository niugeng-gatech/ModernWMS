CREATE TABLE [dbo].[new_supplier] (
 id INT IDENTITY(1,1) PRIMARY KEY,
    supplier_name NVARCHAR(MAX) NOT NULL,
    city NVARCHAR(MAX) NOT NULL,
    address NVARCHAR(MAX) NOT NULL,
    email NVARCHAR(MAX) NOT NULL,
    manager NVARCHAR(MAX) NOT NULL,
    contact_tel NVARCHAR(MAX) NOT NULL,
    creator NVARCHAR(MAX) NOT NULL,
    create_time datetime2 DEFAULT GETDATE(),
    last_update_time datetime2 DEFAULT GETDATE(),
    is_valid bit NOT NULL,
    tenant_id bigint NOT NULL
);

INSERT INTO [dbo].[new_supplier] (supplier_name, city, address, email, manager, contact_tel, creator, create_time, last_update_time, is_valid, tenant_id)
SELECT supplier_name, city, address, email, manager, contact_tel, creator, create_time, last_update_time, is_valid, tenant_id
FROM [dbo].[supplier] ORDER BY id ASC;

-- Step 3: Drop the old table
DROP TABLE [dbo].[supplier];

-- Step 4: Rename the new table to match the old table's name
