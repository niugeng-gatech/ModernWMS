CREATE TABLE [dbo].[new_category] (
 id INT IDENTITY(1,1) PRIMARY KEY,
 category_name NVARCHAR(MAX) NOT NULL,
    parent_id INT NOT NULL,
    creator NVARCHAR(MAX) NOT NULL,
    create_time datetime2 DEFAULT GETDATE(),
    last_update_time datetime2 DEFAULT GETDATE(),
    is_valid bit NOT NULL,
    tenant_id bigint NOT NULL
);

-- Step 2: Copy data from the old table to the new table
INSERT INTO [dbo].[new_category] (category_name, parent_id, creator, create_time, last_update_time, is_valid, tenant_id)
SELECT category_name, parent_id, creator, create_time, last_update_time, is_valid, tenant_id
FROM [dbo].[category];

-- Step 3: Drop the old table
DROP TABLE [dbo].[category];

-- Step 4: Rename the new table to match the old table's name
EXEC sp_rename '[dbo].[new_category]', 'category';
